export default {
  async fetch(request, env) {
    const cors = {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type, X-API-Key',
    };
    if (request.method === 'OPTIONS')
      return new Response(null, { headers: cors });

    const key = request.headers.get('X-API-Key');
    if (key !== env.API_KEY)
      return new Response('Unauthorized', { status: 401, headers: cors });

    const url = new URL(request.url);
    const path = url.pathname;

    // ── STATE (KV) ──────────────────────────────
    if (path === '/state') {
      if (request.method === 'GET') {
        const data = await env.HIKING_KV.get('journal_state');
        return new Response(data || 'null', {
          headers: { ...cors, 'Content-Type': 'application/json' }
        });
      }
      if (request.method === 'POST') {
        const body = await request.text();
        await env.HIKING_KV.put('journal_state', body);
        return new Response('{"ok":true}', { headers: cors });
      }
    }

    // ── IMAGES (R2) ─────────────────────────────
    const imgMatch = path.match(/^\/img\/([a-z0-9]+)$/i);
    if (imgMatch) {
      const imgId = imgMatch[1];
      if (request.method === 'GET') {
        const obj = await env.HIKING_R2.get(imgId);
        if (!obj) return new Response('Not found', { status: 404, headers: cors });
        return new Response(obj.body, {
          headers: { ...cors, 'Content-Type': obj.httpMetadata?.contentType || 'image/jpeg' }
        });
      }
      if (request.method === 'PUT') {
        const ct = request.headers.get('Content-Type') || 'image/jpeg';
        await env.HIKING_R2.put(imgId, request.body, { httpMetadata: { contentType: ct } });
        return new Response('{"ok":true}', { headers: cors });
      }
      if (request.method === 'DELETE') {
        await env.HIKING_R2.delete(imgId);
        return new Response('{"ok":true}', { headers: cors });
      }
    }

    return new Response('Not found', { status: 404, headers: cors });
  }
};
